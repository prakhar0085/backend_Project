// user.routes.js ya video.routes.js mein
import express from "express";
const router = express.Router();

router.post("/upload", async (req, res) => {
    try {
        // Express-fileupload automatically files ko req.files mein daalta hai
        if (!req.files || !req.files.video) {
            return res.status(400).json({
                success: false,
                message: "Koi file nahi mili"
            });
        }

        // File mil gayi temporary storage se
        const videoFile = req.files.video;
        
        console.log("Temporary file info:", {
            name: videoFile.name,
            size: videoFile.size,
            tempFilePath: videoFile.tempFilePath, // Ye important hai!
            mimetype: videoFile.mimetype
        });

        // Ab Cloudinary pe upload karo
        // videoFile.tempFilePath use karo actual file ke liye
        
        // Example:
        const cloudinaryResult = await cloudinary.uploader.upload(
            videoFile.tempFilePath, {
                resource_type: "video"
            }
        );

        // Temporary file automatically delete ho jayegi
        // Express-fileupload khud handle karta hai cleanup

        res.json({
            success: true,
            message: "Upload successful!",
            data: cloudinaryResult
        });

    } catch (error) {
        console.error("Upload error:", error);
        res.status(500).json({
            success: false,
            message: "Upload fail ho gaya",
            error: error.message
        });
    }
});
